287def load_lda_config(cfg_fn, jar='default'):
288    """
289    Load the lda config file from a json file.
290
291    Args
292
293        cfg_fn: path to json config file
294
295    Returns
296
297        config object
298    """
299    # load jaon cfg declaration  as a dict
300    with open(cfg_fn, 'r') as inf:
301        j = json.load(inf)
302
303    if "Beta" not in j or j["Beta"] is None:
304        j["pclda_config"]["Beta"] = j["pclda_config"]["NoTopics"] / 50
305
306    # Initialize SimpleLDAConfiguration
307    slc = new_simple_lda_config(jar_dict=j["jars"], jar=jar)
308    print(type(slc), slc)
309    java_types = _get_java_types()
310
311    # replace slc init values with dict
312    for k, v in j["pclda_config"].items():
313        method_name = f"set{k}"
314        method = getattr(slc, method_name, None)
315        if method:
316            if k in java_types:
317                method(java_types[k](v))
318            else:
319                method(v)
320        else:
321            warnings.warn(f"Unrecognized key in provided config file :: {k} = {v}")
322    print(type(slc), slc)

336def load_lda_sampler(ldaconfig, ds, store_dir="stored_samplers"):

363    # Initialize StringClassArrayIterator for the training data

395                 testset=None, save_sampler=True):
396    # Cast the LDA configuration
397    lcfg = jpype.JObject(ldaconfig, "cc.mallet.configuration.LDAConfiguration")

400    try:
401        lda = jpype.JClass(sampler_type)(lcfg)
402    except jpype.JException as ex:
403        lda = ex  # Handle the exception if the sampler creation fails
404
405    # Add instances to the sampler

407
408    # If test data is provided, add it to the sampler
409    if testset is not None:
410        lda.addTestInstances(testset)
411
412    print("**********************", iterations)
413
414    # Perform sampling
415    print("********************", iterations, type(iterations))
416    try:
417        lda.sample(iterations)
418    except Exception as e:
419        print("EXCEPTION in lda.sample():", e)
420    # If we need to save the sampler, perform the saving procedure
421    if save_sampler:

423        sampler_folder = ldaconfig.getSavedSamplerDirectory(sampler_dir)

425        util = jpype.JClass("cc.mallet.util.LDAUtils")()
